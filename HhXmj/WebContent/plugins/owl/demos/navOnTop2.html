Navigation moved on top by afterInit callback

  • Javascript
  • HTML
  • CSS
$(document).ready(function() {

  $("#owl-demo").owlCarousel({
    navigation:true,
    afterInit : function(elem){
      var that = this
      that.owlControls.prependTo(elem)
    }
  });

});
